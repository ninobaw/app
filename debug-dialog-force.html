🔧 Debug Dialogue Force - AeroDoc

📋 Instructions

1. Connectez-vous à AeroDoc dans un autre onglet

2. Allez dans Paramètres > Tags

3. Ouvrez la console du navigateur (F12)

4. Exécutez les commandes ci-dessous dans la console

🚀 Commandes de Debug

💻 Commandes Console

1. Forcer l'ouverture du dialogue :

// Forcer l'ouverture du dialogue window.forceOpenTagDialog = () => { console.log('🔧 Forcing tag dialog to open...'); // Chercher le bouton "Nouveau tag" const button = document.querySelector('button:has(svg + span)') || Array.from(document.querySelectorAll('button')).find(b => b.textContent?.includes('Nouveau tag')); if (button) { console.log('✅ Found "Nouveau tag" button, clicking...'); button.click(); // Vérifier après 1 seconde setTimeout(() => { const dialog = document.querySelector('[role="dialog"]'); console.log('Dialog found after click:', !!dialog); if (dialog) { console.log('Dialog content:', dialog.innerHTML.substring(0, 200) + '...'); } }, 1000); } else { console.log('❌ "Nouveau tag" button not found'); console.log('Available buttons:', Array.from(document.querySelectorAll('button')).map(b => b.textContent)); } }; // Exécuter window.forceOpenTagDialog();

2. Vérifier l'état du dialogue :

// Vérifier l'état actuel window.checkDialogState = () => { console.log('🔍 Checking dialog state...'); const dialog = document.querySelector('[role="dialog"]'); const overlay = document.querySelector('[data-radix-dialog-overlay]'); const content = document.querySelector('[data-radix-dialog-content]'); console.log('Dialog element:', !!dialog); console.log('Overlay element:', !!overlay); console.log('Content element:', !!content); if (dialog) { console.log('Dialog visible:', dialog.style.display !== 'none'); console.log('Dialog classes:', dialog.className); } // Vérifier les logs récents console.log('Recent SettingsPage logs should show above...'); }; // Exécuter window.checkDialogState();

3. Surveiller les changements :

// Surveiller les changements du DOM window.watchDialogChanges = () => { console.log('👀 Starting dialog DOM watcher...'); const observer = new MutationObserver((mutations) => { mutations.forEach((mutation) => { if (mutation.type === 'childList') { mutation.addedNodes.forEach((node) => { if (node.nodeType === 1 && ( node.matches('[role="dialog"]') || node.querySelector('[role="dialog"]') )) { console.log('🆕 Dialog added to DOM:', node); } }); mutation.removedNodes.forEach((node) => { if (node.nodeType === 1 && ( node.matches('[role="dialog"]') || node.querySelector('[role="dialog"]') )) { console.log('🗑️ Dialog removed from DOM:', node); console.trace('Dialog removal stack trace:'); } }); } }); }); observer.observe(document.body, { childList: true, subtree: true }); console.log('✅ DOM watcher started. Try opening the dialog now...'); // Arrêter après 30 secondes setTimeout(() => { observer.disconnect(); console.log('🛑 DOM watcher stopped'); }, 30000); }; // Exécuter window.watchDialogChanges();

4. Empêcher la fermeture :

// Empêcher toute fermeture de dialogue window.preventDialogClose = () => { console.log('🛡️ Setting up dialog close prevention...'); // Intercepter tous les événements de fermeture document.addEventListener('keydown', (e) => { if (e.key === 'Escape') { console.log('🚫 Prevented Escape key'); e.preventDefault(); e.stopPropagation(); } }, true); document.addEventListener('click', (e) => { const dialog = document.querySelector('[role="dialog"]'); if (dialog && !dialog.contains(e.target)) { console.log('🚫 Prevented outside click'); e.preventDefault(); e.stopPropagation(); } }, true); console.log('✅ Dialog close prevention active'); }; // Exécuter window.preventDialogClose();

📊 Analyse des Logs

Après avoir exécuté les commandes, analysez les logs pour identifier :

  • Montage/Démontage : "Component mounted/unmounted"
  • Changements d'état : "createTagDialogOpen changed to"
  • Rendus : "Rendering with open ="
  • Fermetures : "Dialog was closed, stack trace"
