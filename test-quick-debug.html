🔧 Debug Rapide - Tags

Instructions :

1. Allez dans AeroDoc > Paramètres > Tags

2. Ouvrez la console (F12)

3. Copiez et exécutez cette commande :

// Test rapide du dialogue tags console.log('🔍 DEBUT DEBUG RAPIDE'); // 1. Vérifier les permissions const user = JSON.parse(localStorage.getItem('user') || '{}'); console.log('👤 Utilisateur:', user.email, 'Rôle:', user.role); // 2. Vérifier si on est sur la bonne page const currentUrl = window.location.href; console.log('📍 URL actuelle:', currentUrl); // 3. Chercher le bouton "Nouveau tag" const buttons = Array.from(document.querySelectorAll('button')); const tagButton = buttons.find(b => b.textContent?.includes('Nouveau tag')); console.log('🔘 Bouton "Nouveau tag" trouvé:', !!tagButton); if (tagButton) { console.log('🔘 Texte du bouton:', tagButton.textContent); console.log('🔘 Classes du bouton:', tagButton.className); // 4. Tester le clic console.log('👆 Test du clic...'); tagButton.click(); // 5. Vérifier après le clic setTimeout(() => { const dialog = document.querySelector('[role="dialog"]'); console.log('💬 Dialogue présent après clic:', !!dialog); if (!dialog) { console.log('❌ PROBLEME: Le dialogue ne s\'ouvre pas'); console.log('🔍 Vérifiez les logs SettingsPage et TagManagementSection ci-dessus'); } else { console.log('✅ SUCCES: Le dialogue s\'ouvre !'); } }, 500); } else { console.log('❌ PROBLEME: Bouton "Nouveau tag" non trouvé'); console.log('🔍 Boutons disponibles:', buttons.map(b => b.textContent?.trim()).filter(Boolean)); } console.log('🔍 FIN DEBUG RAPIDE');
